Alpine.js 슬라이더 컴포넌트 예제

이 예제는 Alpine.js를 사용하여 이미지 슬라이더 컴포넌트를 구현하고 번들링한 결과입니다.

아래 예제는 슬라이더 컴포넌트를 사용하는 방법을 보여줍니다.

이미지 슬라이더

사용 방법

필수 파일 로드

            
<!-- 로드: Alpine.js -> 슬라이더 스타일 -> 슬라이더 스크립트 -->
<script defer src="https://unpkg.com/alpinejs@3.x.x/dist/cdn.min.js"></script>
<link rel="stylesheet" href="./dist/slider-styles.css">
<script src="./dist/slider-bundle.js"></script>

마운트

            
// 사용자 지정 옵션으로 마운트
const sliderConfig = {
    transitionType: 'slide',  // 화면 전환: slide, fade
    autoplayEnabled: true,    // 자동 화면 전환
    interval: 3,              // 화면 전환 간격 (초)
    transitionDuration: 0.6,  // 전환 대기 시간 (초)
    imagePosition: 'center',  // 이미지 표시 기준 (top, center, bottom)
    showIndicators: true,     // 현재 이미지 순서 표시
    showControl: true,        // 제어기 표시
    images: [
        { src: "img1.jpg", alt: "내 이미지 1", href: "http://www.example.com" },
        { src: "img2.jpg", alt: "내 이미지 2", href: "http://www.example.org" }
    ]
};
const sliderCustom = SliderComponent.registerSlider(sliderConfig);
